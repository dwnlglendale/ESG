CREATE VIEW [dbo].[User]
AS
SELECT u.UserId, u.Username, u.Password, u.FullName AS Name, u.T24UserName, u.Email, CASE u.IsActive WHEN 1 THEN 'Yes' WHEN 0 THEN 'No' END AS Authorized, u.BranchId, u.ChangePassword, u.IsActiveDirectoryUser, u.IsLocked, u.LogInAttemptCount, u.LastFailedLogInAttempt, 
             u.ClientId, '' AS Globus_ID, b.TypeId AS TypeId, b.BranchName
FROM   UserProfileServiceDb.dbo.Users AS u INNER JOIN
             UserProfileServiceDb.dbo.Branches AS b ON b.Id = u.BranchId

	group by BranchName,StatusName
	order by BranchName
